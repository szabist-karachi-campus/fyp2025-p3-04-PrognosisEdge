def plot_target_distribution(target, title, xlabel, ylabel):
    
    # Calculate class counts
    class_counts = pd.Series(target).value_counts().sort_index()

    # Plot the distribution
    plt.figure(figsize=(10, 6))
    bars = plt.bar(class_counts.index, class_counts.values, color="skyblue", alpha=0.8)

    # Add data labels dynamically above the bars
    for bar in bars:
        height = bar.get_height()
        plt.text(
            bar.get_x() + bar.get_width() / 2, 
            height + height * 0.02,  # Offset based on bar height (2% higher)
            f"{int(height)}", 
            ha="center", 
            va="bottom", 
            fontsize=10
        )

    # Set title and labels
    plt.title(title, fontsize=14, fontweight="bold")
    plt.xlabel(xlabel, fontsize=12)
    plt.ylabel(ylabel, fontsize=12)
    plt.xticks(fontsize=10)
    plt.yticks(fontsize=10)
    plt.tight_layout()
    plt.show()
